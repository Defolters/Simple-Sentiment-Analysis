#    ,  
def tokenize_text(text):
    tokens = nltk.word_tokenize(text)
    tokens = [token.strip() for token in tokens]
    return tokens


#   
def remove_special_characters(text):
    tokens = tokenize_text(text)
    avoid = string.punctuation + ''
    avoid = avoid.replace('(', '')
    avoid = avoid.replace(')', '')
    avoid += string.ascii_lowercase
    avoid += string.digits
    #   
    pattern = re.compile('[{}]'.format(avoid))
    # sub      pattern  ' '
    filtered_text = ' '.join([pattern.sub(' ', token) for token in tokens])
    return filtered_text


#  -  ,    
def remove_stopwords(text):
    tokens = tokenize_text(text)
    filtered_tokens = [token for token in tokens if (token not in stopword_list) \
                       & ((len(token) > 3) | (token in ['(', ')']))]
    filtered_text = ' '.join(filtered_tokens)
    return filtered_text

def estimating_tweets(estimated_reviews, border):
    bad = 0

    positive = 0
    for review in estimated_reviews:
        label = 0

        for word in review[1].split(' '):
            label += words_with_estimation[word][1]
            number += 1
        label /= number
        if label > border:
            label = 1
            positive += 1
        elif label < -border:
            label = -1
            bad += 1
        else:
            label = 0
            neutral += 1
        review.append(label)

def plus_ten(time):
    hour = int(time[0])
    minute = int(time[1])
    if minute>=50:
        hour += 1
        hour %= 24
        minute += 10
        minute %= 60
    else:
        minute += 10
    
    minute = str(minute)
    if len(minute) < 2:
        minute = '0' + minute
    hour = str(hour)
    if len(hour) < 2:
        hour = '0' + hour
    
    return [hour, minute]
